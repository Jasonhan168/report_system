# ─── 数据库配置（必填）─────────────────────────────────────────────────────────
# MySQL 连接字符串格式：mysql://用户名:密码@主机:端口/数据库名
DATABASE_URL=mysql://root:your_password@localhost:3306/report_db

# ─── 认证模式（必填）──────────────────────────────────────────────────────────
# local  = 本地账号密码登录（测试/开发环境推荐）
# ldap   = AD域 LDAP 登录（生产环境推荐）
AUTH_MODE=local

# ─── JWT 密钥（必填，生产环境请使用强随机字符串）──────────────────────────────
# 生成方法（在项目目录执行）：
#   node -e "console.log(require('crypto').randomBytes(32).toString('hex'))"
JWT_SECRET=change-this-to-a-strong-random-secret-in-production

# ─── 服务端口（可选，默认 3000）────────────────────────────────────────────────
PORT=3000
